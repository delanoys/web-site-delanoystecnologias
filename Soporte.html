Copiar archivos desde Cmd

@echo off
setlocal EnableDelayedExpansion

:: Obtener la fecha actual en formato AAAAMMDD
set "fecha=%date:~6,4%%date:~3,2%%date:~0,2%"

:: Definir rutas de origen y destino
set "origen=C:\Program Files (x86)\Apcontabilidad"
set "destino=D:\Backup\Apcontabilidad_%fecha%"

:: Crear directorio de destino si no existe
if not exist "%destino%" mkdir "%destino%"

:: Copiar todos los archivos del origen al destino
xcopy "%origen%\*.*" "%destino%\" /E /H /C /I /Y

:: Verificar si la copia fue exitosa
if %ERRORLEVEL% equ 0 (
echo Copia realizada con exito en %destino%
) else (
echo Error al realizar la copia
)

pause
endlocal
